        import itertools
        return itertools.permutations(nums)

## Solution II
backtracking

```python
class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        res = []
        def backtrack(nums, pre_list):
            if len(nums) <= 0:
                res.append(pre_list)
            else:
                for i in nums:
                    p_list = pre_list.copy()
                    p_list.append(i)
                    left_nums = nums.copy()
                    left_nums.remove(i)
                    backtrack(left_nums, p_list)
        backtrack(nums, [])
        return res

# another version
class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        res = []
        length = len(nums)
        def backtrack(start = 0):
            if start == length:
                res.append(nums[:])  # nums[:] will return a copy of nums
            for i in range(start, length):
                nums[start], nums[i] = nums[i], nums[start]
                backtrack(start + 1)
                nums[start], nums[i] = nums[i], nums[start]
        backtrack()
        return res
```

#### Complexity Analysis:
- Time: O(N!)
- Space: O(N)
